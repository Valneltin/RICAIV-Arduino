/* Hall_Grove est un programme qui allume une led lorsque le capteur à effet Hall détecte la présence d'un aimant*/
                                 
#define capteur 2                                     // affectation des broches  
#define led 3


void setup()
{
  pinMode(led, OUTPUT);                         // la broche led est en sortie
  pinMode(capteur, INPUT);                     // la broche capteur est en entrée
}

void loop()
{
  if(digitalRead(capteur)==LOW)             // si l'aimant est présent devant le capteur
  {
    digitalWrite(led,HIGH);                        // on allume la led
  }
  else
  {
    digitalWrite(led,LOW);                         // on éteint la led
  }
}
